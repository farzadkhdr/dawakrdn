        // نیشاندانی پێشنیارەکانی شوێنەکان
        const locationInput = document.getElementById('locationInput');
        const locationSuggestions = document.getElementById('locationSuggestions');
        
        locationInput.addEventListener('input', function() {
            const inputValue = this.value.toLowerCase();
            locationSuggestions.innerHTML = '';
            
            if (inputValue.length > 0) {
                const filteredLocations = locations.filter(location => 
                    location.toLowerCase().includes(inputValue)
                );
                
                if (filteredLocations.length > 0) {
                    filteredLocations.forEach(location => {
                        const div = document.createElement('div');
                        div.className = 'suggestion-item';
                        div.textContent = location;
                        div.addEventListener('click', function() {
                            locationInput.value = this.textContent;
                            locationSuggestions.style.display = 'none';
                        });
                        locationSuggestions.appendChild(div);
                    });
                    
                    locationSuggestions.style.display = 'block';
                } else {
                    locationSuggestions.style.display = 'none';
                }
            } else {
                locationSuggestions.style.display = 'none';
            }
        });
        
        // داخستنی پێشنیارەکان کاتێ کلیک لەسەر شوێنێکی تر دەکرێت
        document.addEventListener('click', function(e) {
            if (!locationInput.contains(e.target) && !locationSuggestions.contains(e.target)) {
                locationSuggestions.style.display = 'none';
            }
        });
        
        // ناردنی داواکاریەکە
        document.getElementById('houseSellingForm').addEventListener('submit', async function(e) {
            e.preventDefault();
            
            // چالاککردنی بارە
            showLoader(true);
            
            // وەرگرتنی زانیارییەکان لە فۆرمەکە
            const formData = {
                name: document.getElementById('sellerName').value,
                phone: document.getElementById('sellerPhone').value,
                location: document.getElementById('locationInput').value,
                sellingType: document.getElementById('sellingType').value,
                size: document.getElementById('houseSize').value + ' م²',
                price: document.getElementById('housePrice').value + ' دینار',
                type: 'فرۆشتی خانوو',
                date: new Date().toLocaleString('ar-IQ'),
                timestamp: new Date().toISOString()
            };
            
            // ناردن بۆ API
            await sendRequestToAPI(formData);
        });
        
        // چالاککردن/لەکارخستنی بارە
        function showLoader(show) {
            const loader = document.getElementById('formLoader');
            const submitBtn = document.getElementById('submitBtn');
            const submitText = document.getElementById('submitText');
            
            if (show) {
                loader.style.display = 'block';
                submitBtn.disabled = true;
                submitText.textContent = 'ناردن...';
                submitBtn.style.opacity = '0.7';
            } else {
                loader.style.display = 'none';
                submitBtn.disabled = false;
                submitText.textContent = 'ناردنی داواکاری';
                submitBtn.style.opacity = '1';
            }
        }
        
        // ناردن بۆ API
        async function sendRequestToAPI(formData) {
            try {
                console.log('ناردنی داواکاری بۆ API:', formData);
                
                const response = await fetch('/api/submit-request', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify(formData)
                });
                
                console.log('وەڵامی API:', response.status, response.statusText);
                
                const result = await response.json();
                console.log('داتاکانی وەڵام:', result);
                
                // لەکارخستنی بارە
                showLoader(false);
                
                if (result.success) {
                    // نیشاندانی ژمارەی داواکاری
                    if (result.systemResponse && result.systemResponse.request && result.systemResponse.request.id) {
                        document.getElementById('requestId').textContent = 
                            `ژمارەی داواکاری: ${result.systemResponse.request.id}`;
                        document.getElementById('requestId').style.display = 'block';
                    }
                    showSuccessMessage();
                } else {
                    showErrorMessage(result.message || 'کێشەیەک ڕوویدا لە ناردنی داواکاری');
                }
            } catch (error) {
                console.error('هەڵە لە ناردنی داواکاری:', error);
                
                // لەکارخستنی بارە
                showLoader(false);
                
                showErrorMessage('پەیوەندی لەگەڵ سێرڤەرەکەدا شکستی هێنا. تکایە دووبارە هەوڵ بدەرەوە.');
            }
        }

        function showSuccessMessage() {
            document.getElementById('sellHouseForm').style.display = 'none';
            document.getElementById('mainSections').style.display = 'none';

            document.getElementById('mainSections').style.display = 'none';

            document.getElementById('errorMessage').style.display = 'block';
        }
